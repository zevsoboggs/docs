## Заголовки

<ParamField header="X-Api-Key" type="string" required>
  Ваш API ключ
</ParamField>

## Query параметры

<ParamField query="page" type="integer" default="1">
  Номер страницы
</ParamField>

<ParamField query="pageSize" type="integer" default="20">
  Размер страницы (максимум 100)
</ParamField>

<ParamField query="status" type="string">
  Фильтр по статусу: `Pending`, `PartiallyPaid`, `Paid`, `Overpaid`, `Expired`, `Cancelled`
</ParamField>

<ParamField query="currency" type="string">
  Фильтр по валюте: `USDT_TRC20`, `USDT_ERC20`, `USDC_ERC20`, `USDT_BEP20`, `USDC_BEP20`
</ParamField>

<ParamField query="fromDate" type="string">
  Начало периода (ISO 8601), например `2026-01-01T00:00:00Z`
</ParamField>

<ParamField query="toDate" type="string">
  Конец периода (ISO 8601), например `2026-01-31T23:59:59Z`
</ParamField>
