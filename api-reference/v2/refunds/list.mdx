---
title: 'Возвраты по счёту'
description: 'Список всех заявок на возврат по конкретному счёту'
api: 'GET https://loveandpay.io/api/v2/invoices/{id}/refund'
---

Возвращает **все** заявки на возврат по счёту (в порядке от новых к старым). Полезно,
чтобы посчитать уже возвращённую сумму и остаток при частичных возвратах.

## Заголовки

<ParamField header="x-api-key" type="string" required>
  Ваш API ключ
</ParamField>

<ParamField header="x-timestamp" type="string" required>
  Unix timestamp в миллисекундах
</ParamField>

<ParamField header="x-signature" type="string" required>
  HMAC-SHA256 подпись
</ParamField>

## Path параметры

<ParamField path="id" type="string" required>
  UUID счёта или его номер (`INV-...`)
</ParamField>

```javascript JavaScript
const crypto = require('crypto');

const API_KEY = 'pk_live_xxx';
const SECRET_KEY = 'sk_live_xxx';

const invoiceId = 'INV-260716-0042';
const timestamp = Date.now().toString();
const method = 'GET';
const path = `/api/v2/invoices/${invoiceId}/refund`;

const bodyHash = crypto.createHash('sha256').update('').digest('hex');
const message = `${method}${path}${timestamp}${bodyHash}`;
const signature = crypto.createHmac('sha256', SECRET_KEY).update(message).digest('hex');

const response = await fetch('https://loveandpay.io' + path, {
  method,
  headers: {
    'x-api-key': API_KEY,
    'x-timestamp': timestamp,
    'x-signature': signature
  }
});

console.log(await response.json());
```

```python Python
import hashlib
import hmac
import time
import requests

API_KEY = 'pk_live_xxx'
SECRET_KEY = 'sk_live_xxx'

invoice_id = 'INV-260716-0042'
timestamp = str(int(time.time() * 1000))
method = 'GET'
path = f'/api/v2/invoices/{invoice_id}/refund'

body_hash = hashlib.sha256(''.encode()).hexdigest()
message = f"{method}{path}{timestamp}{body_hash}"
signature = hmac.new(SECRET_KEY.encode(), message.encode(), hashlib.sha256).hexdigest()

response = requests.get(
    'https://loveandpay.io' + path,
    headers={
        'x-api-key': API_KEY,
        'x-timestamp': timestamp,
        'x-signature': signature
    }
)

print(response.json())
```
