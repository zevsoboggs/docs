```javascript JavaScript
const crypto = require('crypto');

const API_KEY = 'pk_live_xxx';
const SECRET_KEY = 'sk_live_xxx';

const timestamp = Date.now().toString();
const method = 'GET';
const path = '/api/v2/webhooks';

const bodyHash = crypto.createHash('sha256').update('').digest('hex');
const message = `${method}${path}${timestamp}${bodyHash}`;
const signature = crypto.createHmac('sha256', SECRET_KEY).update(message).digest('hex');

const response = await fetch('https://loveandpay.io' + path, {
  method,
  headers: {
    'x-api-key': API_KEY,
    'x-timestamp': timestamp,
    'x-signature': signature
  }
});

const { webhooks } = await response.json();

webhooks.forEach(webhook => {
  console.log(`${webhook.url} - события: ${webhook.events.join(', ')}`);
});
```

```python Python
import hashlib
import hmac
import time
import requests

API_KEY = 'pk_live_xxx'
SECRET_KEY = 'sk_live_xxx'

timestamp = str(int(time.time() * 1000))
method = 'GET'
path = '/api/v2/webhooks'

body_hash = hashlib.sha256(''.encode()).hexdigest()
message = f"{method}{path}{timestamp}{body_hash}"
signature = hmac.new(SECRET_KEY.encode(), message.encode(), hashlib.sha256).hexdigest()

response = requests.get(
    'https://loveandpay.io' + path,
    headers={
        'x-api-key': API_KEY,
        'x-timestamp': timestamp,
        'x-signature': signature
    }
)

data = response.json()
for webhook in data['webhooks']:
    print(f"{webhook['url']} - события: {', '.join(webhook['events'])}")
```
