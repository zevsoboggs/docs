export const StatusPage = () => {
  const [data, setData] = React.useState(null);
  const [loading, setLoading] = React.useState(true);
  const [error, setError] = React.useState(null);

  React.useEffect(() => {
    fetch('https://lnp-status.grekdev.com/api/Status')
      .then(r => r.json())
      .then(json => {
        setData(json);
        setLoading(false);
      })
      .catch(err => {
        setError(err.message);
        setLoading(false);
      });
  }, []);

  if (loading) return (
    <div style={{ padding: '40px', textAlign: 'center' }}>
      <div style={{ fontSize: '24px', marginBottom: '8px' }}>‚è≥</div>

      –û—à–∏–±–∫–∞ –∑–∞–≥—Ä—É–∑–∫–∏: {error}
    </div>
  );

  if (!data) return null;

  const statusLabels = {
    1: 'Operational',
    2: 'Degraded',
    3: 'Partial Outage',
    4: 'Major Outage',
    5: 'Maintenance'
  };

  const statusColors = {
    1: '#22c55e',
    2: '#eab308',
    3: '#f97316',
    4: '#ef4444',
    5: '#6b7280'
  };

  const statusIcons = {
    1: 'üü¢',
    2: 'üü°',
    3: 'üü†',
    4: 'üî¥',
    5: 'üîß'
  };

  const overall = data.overallStatus || {};
  const overallStatus = overall.status || 1;

  return (
    <div>
      <div style={{
        padding: '24px',
        borderRadius: '12px',
        marginBottom: '24px',
        border: '2px solid ' + statusColors[overallStatus],
        background: statusColors[overallStatus] + '15'
      }}>
        <div style={{ display: 'flex', alignItems: 'center', gap: '16px' }}>
          <span style={{ fontSize: '36px' }}>{overallStatus === 1 ? '‚úÖ' : '‚ö†Ô∏è'}</span>
          <div>
            <div style={{ fontWeight: '700', fontSize: '20px', marginBottom: '4px' }}>
              {data.statusMessage}
            </div>
            <div style={{ fontSize: '14px', opacity: 0.8 }}>
              <span style={{ color: '#22c55e' }}>‚óè</span> {overall.operationalCount || 0} —Ä–∞–±–æ—Ç–∞—é—Ç
              {(overall.degradedCount || 0) > 0 && <span> ¬∑ <span style={{ color: '#eab308' }}>‚óè</span> {overall.degradedCount} —Å –ø—Ä–æ–±–ª–µ–º–∞–º–∏</span>}
              {(overall.outageCount || 0) > 0 && <span> ¬∑ <span style={{ color: '#ef4444' }}>‚óè</span> {overall.outageCount} –Ω–µ–¥–æ—Å—Ç—É–ø–Ω—ã</span>}
            </div>
          </div>
        </div>
      </div>

      {data.activeIncidents && data.activeIncidents.length > 0 && (
        <div style={{ marginBottom: '24px' }}>
          <h3>üö® –ê–∫—Ç–∏–≤–Ω—ã–µ –∏–Ω—Ü–∏–¥–µ–Ω—Ç—ã ({data.activeIncidents.length})</h3>
          {data.activeIncidents.map((incident, i) => (
            <div key={i} style={{
              padding: '16px',
              borderRadius: '8px',
              marginBottom: '8px',
              background: 'rgba(239,68,68,0.1)',
              borderLeft: '4px solid #ef4444'
            }}>
              <div style={{ fontWeight: '600', marginBottom: '6px' }}>{incident.title}</div>
              <div style={{ fontSize: '14px', opacity: 0.8, marginBottom: '8px' }}>{incident.description}</div>
              <div style={{ fontSize: '12px', opacity: 0.6 }}>
                –ù–∞—á–∞–ª–æ: {new Date(incident.createdAt).toLocaleString('ru-RU')}
              </div>
            </div>
          ))}
        </div>
      )}

      {data.serviceGroups && data.serviceGroups.map((group, gi) => (
        <div key={gi} style={{ marginBottom: '24px' }}>
          <h3>{group.groupName === 'Production' ? 'üöÄ' : '‚öôÔ∏è'} {group.groupName}</h3>
          <div style={{ border: '1px solid var(--border-color, #e5e7eb)', borderRadius: '8px', overflow: 'hidden' }}>
            {group.services && group.services.map((service, si) => (
              <div key={si} style={{
                display: 'flex',
                justifyContent: 'space-between',
                alignItems: 'center',
                padding: '12px 16px',
                borderBottom: si < group.services.length - 1 ? '1px solid var(--border-color, #e5e7eb)' : 'none',
                flexWrap: 'wrap',
                gap: '8px'
              }}>
                <div style={{ display: 'flex', alignItems: 'center', gap: '12px' }}>
                  <span>{statusIcons[service.currentStatus] || 'üü¢'}</span>
                  <div>
                    <div style={{ fontWeight: '600' }}>{service.name}</div>
                    {service.url && <div style={{ fontSize: '12px', opacity: 0.5 }}>{service.url}</div>}
                  </div>
                </div>
                <div style={{ display: 'flex', alignItems: 'center', gap: '16px' }}>
                  {service.uptimePercentage24h !== undefined && (
                    <span style={{ fontSize: '13px', opacity: 0.7 }}>
                      {service.uptimePercentage24h.toFixed(1)}% uptime
                    </span>
                  )}
                  {service.lastResponseTimeMs && (
                    <span style={{ fontSize: '13px', opacity: 0.7 }}>{service.lastResponseTimeMs}ms</span>
                  )}
                  <span style={{
                    padding: '4px 12px',
                    borderRadius: '6px',
                    fontSize: '12px',
                    fontWeight: '600',
                    background: statusColors[service.currentStatus] + '20',
                    color: statusColors[service.currentStatus]
                  }}>
                    {statusLabels[service.currentStatus] || 'Unknown'}
                  </span>
                </div>
              </div>
            ))}
          </div>
        </div>
      ))}

      {data.upcomingMaintenances && data.upcomingMaintenances.length > 0 && (
        <div style={{ marginBottom: '24px' }}>
          <h3>üìÖ –ü–ª–∞–Ω–æ–≤—ã–µ —Ä–∞–±–æ—Ç—ã</h3>
          {data.upcomingMaintenances.map((maint, i) => (
            <div key={i} style={{
              padding: '16px',
              borderRadius: '8px',
              marginBottom: '8px',
              background: 'rgba(59,130,246,0.1)',
              borderLeft: '4px solid #3b82f6'
            }}>
              <div style={{ fontWeight: '600', marginBottom: '6px' }}>{maint.title}</div>
              <div style={{ fontSize: '14px', opacity: 0.8, marginBottom: '8px' }}>{maint.description}</div>
              <div style={{ fontSize: '12px', opacity: 0.6 }}>
                {new Date(maint.scheduledStart).toLocaleString('ru-RU')} ‚Äî {new Date(maint.scheduledEnd).toLocaleString('ru-RU')}
              </div>
            </div>
          ))}
        </div>
      )}

      <div style={{ textAlign: 'center', fontSize: '13px', opacity: 0.5, paddingTop: '16px', borderTop: '1px solid var(--border-color, #e5e7eb)' }}>
        –û–±–Ω–æ–≤–ª–µ–Ω–æ: {new Date(data.generatedAt || Date.now()).toLocaleString('ru-RU')}
      </div>
    </div>
  );
};
